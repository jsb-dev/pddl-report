  (:action move
    :parameters (agent sq-1-1 sq-1-2)
    :precondition
      (and
        (adj sq-1-1 sq-1-2)
        (not
          (pit sq-1-2)
        )
        (at agent sq-1-1)
        (not
          (dead agent)
        )
        (not
          (wumpus sq-1-2)
        )
      )
    :effect
      (and
        (not
          (at agent sq-1-1)
        )
        (at agent sq-1-2)
        (visited sq-1-2)
        (when
          (pit sq-1-2)
          (dead agent)
        )
        (when
          (wumpus sq-1-2)
          (dead agent)
        )
        (when
          (gold sq-1-2)
          (glitter sq-1-2)
        )
        (when
          (exists ?adj and)
          (breeze sq-1-2)
        )
        (when
          (exists ?adj and)
          (stench sq-1-2)
        )
      )
  )
